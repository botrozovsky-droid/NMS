## 📋 Feature 1: Semantic Search (Embeddings)

### Objective
Replace keyword-only search with semantic search using embeddings.

### Implementation

#### 1.1 Embedding Generation
**File:** `semantic-search.js` (~200 LOC)

```javascript
import axios from 'axios';

// Use text-embedding-004 (cheap, fast)
const EMBEDDING_MODEL = 'text-embedding-004';
const EMBEDDING_API = 'https://generativelanguage.googleapis.com/v1beta/models/text-embedding-004:embedContent';

async function generateEmbedding(text) {
  const response = await axios.post(
    `${EMBEDDING_API}?key=${GEMINI_API_KEY}`,
    {
      content: { parts: [{ text }] }
    }
  );

  return response.data.embedding.values; // 768-dim vector
}

// Pre-compute embeddings during consolidation
async function embedNode(node) {
  const text = `${node.name} ${node.type} ${node.rationale}`;
  node.embedding = await generateEmbedding(text);
  return node;
}
```

#### 1.2 Vector Index
**Library:** `hnswlib-node` (local, no cloud dependencies)

```javascript
import { HierarchicalNSW } from 'hnswlib-node';

// Create index

index.initIndex(maxElements);

// Add vectors

  if (node.embedding) {
    index.addPoint(node.embedding, nodeId);
  }

// Search
const results = index.searchKnn(queryEmbedding, k=10);
```

#### 1.3 Hybrid Search
**File:** `memory-manager.js` (update `queryMemory`)

```javascript
async queryMemory(query, options = {}) {
  const { useSemanticSearch = true, topK = 10 } = options;

  let results = [];

  // 1. Keyword search (fast, precise)
  const keywordResults = this.keywordSearch(query);

  // 2. Semantic search (slower, broader)
  if (useSemanticSearch) {
    const semanticResults = await this.semanticSearch(query, topK);

    // Merge and deduplicate
    results = this.mergeResults(keywordResults, semanticResults);
  } else {
    results = keywordResults;
  }

  // 3. Rank by relevance + recency + confidence
  return this.rankResults(results);
}
```

#### 1.4 Migration
**File:** `migrate-v0.2-to-v0.3.js`

```javascript
  }
```

## 📋 Feature 2: Session-End Mini-Consolidation

### Objective
Consolidate top 10-20 important events at session end (not wait 24h).

### Implementation

#### 2.1 Session Detection
**File:** `session-manager.js` (~150 LOC)

```javascript
class SessionManager {
  constructor() {
    this.currentSession = null;
    this.lastActivity = Date.now();
    this.inactivityThreshold = 30 * 60 * 1000; // 30 min
  }

  // Track activity
  recordActivity() {
    this.lastActivity = Date.now();
  }

  // Check if session ended
  isSessionEnded() {
    const inactive = Date.now() - this.lastActivity;
    return inactive > this.inactivityThreshold;
  }

  // End session and trigger mini-consolidation
  async endSession() {
    if (this.currentSession) {
      await this.miniConsolidation(this.currentSession);
      this.currentSession = null;
    }
  }
}
```

#### 2.2 Mini-Consolidation
**File:** `mini-consolidate.js` (~200 LOC)

```javascript
async function miniConsolidation(sessionId) {
  console.log('🌙 Starting session-end mini-consolidation...');

  // 1. Get top 20 events from this session (by importance)
  const sessionEvents = await getSessionEvents(sessionId);
  const topEvents = sessionEvents
    .sort((a, b) => b.importance - a.importance)
    .slice(0, 20);

  // 2. Quick consolidation (no batching, single pass)
  const analysis = await analyzeWithGemini(topEvents);

  // 3. Update graph
  await updateGraph(analysis);

  // 4. Mark as consolidated
  await markEventsConsolidated(topEvents);

  console.log(`✅ Mini-consolidation done: ${topEvents.length} events`);
}
```

## 📋 Feature 3: Transaction Support

### Objective
Atomic operations with rollback capability (all-or-nothing updates).

### Implementation

#### 3.1 Transaction Manager
**File:** `transaction-manager.js` (~250 LOC)

```javascript
class TransactionManager {
  constructor(dataPath) {
    this.dataPath = dataPath;
    this.backupPath = `${dataPath}.backup`;
    this.inTransaction = false;
  }

  // Start transaction
  async begin() {
    if (this.inTransaction) {
      throw new Error('Transaction already in progress');
    }

    // Create backup
    const data = await fs.readFile(this.dataPath, 'utf-8');
    await fs.writeFile(this.backupPath, data);

    this.inTransaction = true;
    this.startTime = Date.now();
  }

  // Commit transaction
  async commit() {
    if (!this.inTransaction) {
      throw new Error('No transaction in progress');
    }

    // Delete backup (commit successful)
    await fs.unlink(this.backupPath);

    this.inTransaction = false;
    console.log(`✅ Transaction committed (${Date.now() - this.startTime}ms)`);
  }

  // Rollback transaction
  async rollback() {
    if (!this.inTransaction) {
      throw new Error('No transaction in progress');
    }

    // Restore from backup
    const backup = await fs.readFile(this.backupPath, 'utf-8');
    await fs.writeFile(this.dataPath, backup);
    await fs.unlink(this.backupPath);

    this.inTransaction = false;
    console.log(`🔄 Transaction rolled back`);
  }

  // Execute operation with transaction
  async execute(operation) {
    try {
      await this.begin();
      const result = await operation();
      await this.commit();
      return result;
    } catch (error) {
      await this.rollback();
      throw error;
    }
  }
}
```

#### 3.3 Transaction for Critical Operations
**File:** `memory-manager.js` (update)

```javascript
// Any operation that modifies graph
async updateNode(nodeId, updates) {
  await graphTransaction.execute(async () => {
    const graph = await loadGraph();

    if (!graph.nodes[nodeId]) {
      throw new Error(`Node ${nodeId} not found`);
    }

    Object.assign(graph.nodes[nodeId], updates);

    await saveGraph(graph);
  });
}

async deleteNode(nodeId) {
  await graphTransaction.execute(async () => {
    const graph = await loadGraph();

    // Delete node
    delete graph.nodes[nodeId];

    // Delete related edges
    for (const [edgeId, edge] of Object.entries(graph.edges)) {
      if (edge.source === nodeId || edge.target === nodeId) {
        delete graph.edges[edgeId];
      }
    }

    await saveGraph(graph);
  });
}
```
